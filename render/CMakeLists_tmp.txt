cmake_minimum_required(VERSION 3.5.1)
project(render)

link_directories(../core/)
include_directories(../core/)

link_directories(../libs/imgui/)
include_directories(../libs/imgui/)

link_directories(../libs/implot/)
include_directories(../libs/implot/)

add_compile_options(-fvisibility=hidden)

if (POLICY CMP0072)
    cmake_policy (SET CMP0072 OLD)
endif(POLICY CMP0072)

find_package(DART REQUIRED COMPONENTS gui gui-osg utils-urdf collision-bullet CONFIG)
find_package(OpenGL REQUIRED)
find_package(GLUT REQUIRED)
find_package(GLEW REQUIRED)

find_package(PythonLibs REQUIRED)
find_package(pybind11 REQUIRED)

include_directories(${DART_INCLUDE_DIRS})
include_directories(${OPENGL_INCLUDE_DIR})
include_directories(${GLUT_INCLUDE_DIR})
include_directories(${GLEW_INCLUDE_DIR})
include_directories(${PYTHON_INCLUDE_DIR})

file(GLOB srcs "*.h" "*.cpp")
add_executable(render ${srcs})
target_link_libraries(render ${PYTHON_LIBRARIES} mss dart dart-gui-osg GL GLU glfw imgui GLEW pybind11::embed stdc++fs)
